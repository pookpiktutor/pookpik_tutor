+function parseHoursStrToMinutes(val) {
+  if (!val) return 0;
+  
+  if (val instanceof Date) {
+    return val.getHours() * 60 + val.getMinutes();
+  }
+  
+  const s = val.toString().trim();
+  if (s.includes('GMT') || s.includes('1899')) {
+    const match = s.match(/(\d{2}):(\d{2}):(\d{2})/);
+    if (match) {
+      const hrs = parseInt(match[1], 10) || 0;
+      const mins = parseInt(match[2], 10) || 0;
+      return hrs * 60 + mins;
+    }
+  }
+  
+  if (s.includes(':')) {
+    const parts = s.split(':');
+    const hrs = parseInt(parts[0], 10) || 0;
+    const mins = parseInt(parts[1], 10) || 0;
+    return hrs * 60 + mins;
+  }
+  if (s.includes('ชม.') || s.includes('นาที')) {
+    return parseHoursLeftToMinutes(s);
+  }
+  const num = parseFloat(s);
+  if (!isNaN(num)) {
+    return Math.round(num * 60);
+  }
+  return 0;

+function migrateDataLearnTeacherNamesToIDs() {
+  try {
+    const db = getDb();
+    const sheet = db.getSheetByName('Data Learn');
+    if (!sheet) return;
+    
+    const lastRow = sheet.getLastRow();
+    if (lastRow < 2) return;
+    
+    const teachersList = getTeachersDB(null);
+    if (!Array.isArray(teachersList) || teachersList.length === 0) return;
+    
+    const range = sheet.getRange(2, 2, lastRow - 1, 2); // Column B and C
+    const values = range.getValues();
+    
+    let updated = false;
+    const resolveId = function(name) {
+      if (!name) return '';
+      const cleanTarget = name.toString().trim().toLowerCase().replace(/^ครู/, '').trim();
+      if (cleanTarget.startsWith('tutor_')) return name; // Already an ID
+      
+      const match = teachersList.find(t => {
+        const tNick = t.nickname.toLowerCase().trim().replace(/^ครู/, '').trim();
+        const tId = (t.teacherId || '').toLowerCase().trim();
+        return tNick === cleanTarget || tId === cleanTarget;
+      });
+      return (match && match.teacherId) ? match.teacherId : name;
+    };
+    
+    for (let i = 0; i < values.length; i++) {
+      const oldB = values[i][0] ? values[i][0].toString().trim() : '';
+      const oldC = values[i][1] ? values[i][1].toString().trim() : '';
+      
+      const newB = resolveId(oldB);
+      const newC = resolveId(oldC);
+      
+      if (newB !== oldB || newC !== oldC) {
+        values[i][0] = newB;
+        values[i][1] = newC;
+        updated = true;
+      }
+    }
+    
+    if (updated) {
+      range.setValues(values);
+      logActivity('System', 'ไมเกรตข้อมูลครู Data Learn', 'แปลงชื่อครูในตารางสอนเก่าเป็นรหัสประจำตัวสำเร็จ');
+    }
+  } catch (e) {
+    Logger.log('Error in migrateDataLearnTeacherNamesToIDs: ' + e.message);
+  }
+}

+/**
+ * ดึงข้อมูลการใช้ห้องเรียนทั้งหมดเฉพาะวันเสาร์และวันอาทิตย์
+ */
+function getWeekendRoomSchedule(logUser) {
+  if (logUser) checkTeacherBlock(logUser);
+  try {
+    ensureDataLearnMigrated(getDb());
+    const rawData = getSheetRows('Data Learn');
+    const rooms = getRoomsList();
+    const teachersList = getTeachersDB(null);
+    
+    const resolveNick = function(nameOrId) {
+      if (!nameOrId) return '';
+      const cleanVal = nameOrId.toString().trim().toLowerCase();
+      const match = teachersList.find(t => {
+        const tId = (t.teacherId || '').toLowerCase().trim();
+        const tNick = t.nickname.toLowerCase().trim();
+        return (tId !== '' && tId === cleanVal) || tNick === cleanVal;
+      });
+      return match ? match.nickname : nameOrId;
+    };
+    
+    const weekendLogs = [];
+    
+    rawData.forEach((row, idx) => {
+      if (idx === 0) return;
+      if (!row[0] || row[0] === '0') return;
+      
+      const dateStr = cleanSheetDate(row[13]);
+      const dateParsed = parseDateString(dateStr);
+      if (!dateParsed) return;
+      
+      // ดึงเฉพาะวันเสาร์ (6) และวันอาทิตย์ (0)
+      const day = dateParsed.getDay();
+      if (day === 6 || day === 0) {
+        weekendLogs.push({
+          subject: row[0] ? row[0].toString().trim() : '',
+          teacherRegular: resolveNick(row[1]),
+          teacherSub: resolveNick(row[2]),
+          timeStart: row[3] ? row[3].toString().trim() : '',
+          timeEnd: row[4] ? row[4].toString().trim() : '',
+          note: row[5] ? row[5].toString().trim() : '',
+          isPresentLive: parseInt(row[6]) || 0,
+          isPresentOnline: parseInt(row[7]) || 0,
+          isLeave: parseInt(row[8]) || 0,
+          isAbsent: parseInt(row[9]) || 0,
+          isMakeup: parseInt(row[10]) || 0,
+          hours: row[12] ? row[12].toString().trim() : '',
+          date: dateStr,
+          roomBranch: row[14] ? row[14].toString().trim() : '',
+          teacherConfirmed: row[15] ? (parseInt(row[15]) || 0) : 0,
+          numKids: row[16] ? (parseInt(row[16]) || 0) : 0,
+          rowIndex: idx + 1
+        });
+      }
+    });
+    
+    return {
+      rooms: rooms,
+      classes: weekendLogs
+    };
+  } catch (err) {
+    return { error: err.message };
+  }
+}
